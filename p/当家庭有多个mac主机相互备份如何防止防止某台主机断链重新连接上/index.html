当从机Mac变更iP时，通过SMB协议自动上传iP给主机

2024-12-11
阅读时长: 2 分钟

注意，如果创建.sh文件后不可以执行，请执行sudo chmod 755 文件名.sh来修改权限

使用场景: sudo 命令

通过Mac的mount_smbfs 明文方式转入用户名 & 密码 & 待转入主机iP

#!/bin/bash
# 修复版 SMB 传输脚本（解决 MOUNT_POINT 无法访问）
Athor="Benjamin"
Creat_Time=“20241210”
SMB_USER="" 
SMB_PASS=""    
SMB_IP=""
SMB_SHARE="SMB_USER/Desktop"   
# 挂载点改为用户桌面（权限最高，绝对避免访问问题）

NETWORK_INTERFACE="en0"
MOUNT_POINT="$HOME/Desktop/SMB_Temp_Mount"
LOCAL_SRC_PATH="$HOME/Desktop/ip_config.json"  #LOCAL_SRC_PATH="$HOME/Desktop/*.jpg"
LOCAL_NEW_PATH="$HOME/Desktop/ip_config_V.json"
MAX_RETRY=2
RETRY_INTERVAL=5

sync_to_smb() {

if mount | grep -q "^//.* $MOUNT_POINT "; then
    echo "⚠️  卸载残留挂载..."
    umount -f "$MOUNT_POINT"
    sleep 2
fi

rm -rf "$MOUNT_POINT"
mkdir -p "$MOUNT_POINT"
# 赋予挂载点最高读写权限
chmod 777 "$MOUNT_POINT"

retry_count=0
mount_success=0
while [ $retry_count -lt $MAX_RETRY ] && [ $mount_success -eq 0 ]; do
    echo "🔄 第 $((retry_count+1)) 次挂载..."
    # 直接挂载，密码嵌入 URL
    mount_smbfs "//$SMB_USER:$SMB_PASS@$SMB_IP/$SMB_SHARE" "$MOUNT_POINT"
    
    # 双重校验：是否挂载 + 是否可写
    if [ -w "$MOUNT_POINT" ] && mount | grep -q "^//.* $MOUNT_POINT "; then
        echo "✅ 挂载成功，权限正常！"
        mount_success=1
    else
        echo "❌ 挂载失败，${RETRY_INTERVAL}秒后重试..."
        sleep $RETRY_INTERVAL
        retry_count=$((retry_count+1))
    fi
done

if [ $mount_success -eq 1 ]; then
    echo "📤 传输文件中..."
    cp -r $LOCAL_SRC_PATH "$MOUNT_POINT/"
    if [ $? -eq 0 ]; then
        echo "🎉 传输完成！"

        sleep 10

        Cat_LOCAL_iP_json

    else
        echo "❌ 传输失败，请检查本地文件路径"
    fi
    # 卸载 + 清理
    umount -f "$MOUNT_POINT"
    rm -rf "$MOUNT_POINT"
else
    echo "❌ 多次挂载失败，请检查：1. 目标IP/共享名 2. 用户名密码 3. 网络连通性"
    rm -rf "$MOUNT_POINT"
    exit 1
fi

}
init_json() 
{
    current_ip=$(ifconfig $NETWORK_INTERFACE | grep "inet " | grep -v  127.0.0.1  | awk '{print $2}')
    if [[ -z $current_ip ]]; then
        echo "[$(date +'%Y-%m-%d %H:%M:%S')] 获取初始 IP 失败"
        exit 1
    fi
    # 生成 JSON 格式数据
    cat > $LOCAL_SRC_PATH << EOF
{
    "ip_address": "$current_ip",
    "update_time": "$(date +'%Y-%m-%d %H:%M:%S')",
    "interface": "$NETWORK_INTERFACE"
}
EOF
    echo "[$(date +'%Y-%m-%d %H:%M:%S')] 初始化 IP 配置文件：$LOCAL_SRC_PATH"
}

get_current_ip_info() {
    current_ip=$(ifconfig $NETWORK_INTERFACE | grep "inet " | grep -v  127.0.0.1  | awk '{print $2}')
    if [[ -z $current_ip ]]; then
        echo ""
        return 1
    fi
    # 返回 JSON 格式字符串
    echo "{
    \"ip_address\": \"$current_ip\",
    \"update_time\": \"$(date +'%Y-%m-%d %H:%M:%S')\",
    \"interface\": \"$NETWORK_INTERFACE\"
}"
}
