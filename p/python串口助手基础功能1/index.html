class Serial_Assistant:

    #初始化函数
    def __init__(self):
        self.Init_all_varaibles()   #初始化用到的变量
        self.Init_all_timers()    #QT自带的定时器
        self.ui = QUiLoader().load('Serial_Assistant.ui') #使用QUiLoader函数，将UI文件解析成程序能处理的UI文本文件
        self.Init_all_functions()  # 所有的槽函数
        self.Init_serial_settings() #相关串口配置
        self.Init_matplotlib()

    def Init_matplotlib(self):
        # 创建Matplotlib画布
        self.canvas = FigureCanvas(Figure(figsize=(4.8, 0.45)))
        self.axes = self.canvas.figure.add_subplot(111)

        # 确保画布有合适的大小
        self.canvas.setMinimumSize(450, 188)

        # 创建场景并添加到QGraphicsView
        self.scene = QGraphicsScene()
        self.ui.Graph_Show.setScene(self.scene)

        # 将画布添加到场景中
        self.scene.addWidget(self.canvas)

    # 初始化相关标志变量
    def Init_all_varaibles(self):
        self.is_hex_receive = False  #受否是16进制的接收
        self.is_time_stamp = False #时间戳是否开启
        self.is_timing_send = False #定时发送
        self.is_hex_send = False #16进制发送
        self.is_send_newline = False #发送是否新开启
        self.receive_raw_data = bytearray()  # 存储原始接收数据使用 bytearray 格式，转换成字节数组的类型，方便16进制的显示与发送
        self.whole_send_text = bytearray()
        self.time_stamp_time = 1000  #时间输入的定时
        self.timing_send_time = 1000 #定时发送的定时
        self.current_ports = [] #存储窗口选择界面的串口列表

        # 使用队列处理图形数据
        self.graph_data_queue = queue.Queue()
        self.t_list = []
        self.x_list = []
        self.y_list = []
        self.max_data_points = 100  # 最大数据点数
        # 新增数据接收缓冲区
        self.data_buffer = ""  # 用于累积数据
        self.buffer_max_size = 2048  # 缓冲区最大大小
        # 新增数据解析统计
        self.data_points_received = 0
        self.data_points_parsed = 0


    # 初始化用到的定时器
    def Init_all_timers(self):
        # 设置定时器用于定期检查串口变化，看串口增加或减少了，都可以通过这个定时器的任务函数进行实现
        self.timer_of_change_serial = QTimer()
        self.timer_of_change_serial.timeout.connect(self.check_serial_ports)
        self.timer_of_change_serial.start(1000)  # 每1秒检查一次

        #接受任务的定时器
        self.timer_of_receive = QTimer()
        self.timer_of_receive.timeout.connect(self.receive_serial_data_task)
        self.timer_of_receive.start(10) #每10ms进行一次判断，然后接收

        #自动保存的定时器，勾选自动保存之后，每过一秒进行一个保存、刷新
        self.timer_of_auto_save = QTimer()
        self.timer_of_auto_save.timeout.connect(self.timer_of_auto_save_task)
        self.timer_of_auto_save.start(1000) 

        #时间戳与定时器发送相关的定时器
        self.timer_of_time_stamp = QTimer()
        self.timer_of_time_stamp.timeout.connect(self.time_stamp_time_task)

        self.timer_of_timing_send = QTimer()
        self.timer_of_timing_send.timeout.connect(self.Send_Period_Time_Task)

        #当前时间的显示
        self.timer_of_time_refresh = QTimer()
        self.timer_of_time_refresh.timeout.connect(self.time_refresh_show)
        self.timer_of_time_refresh.start(1000)  # 每1秒检查一次

        # 添加图形更新定时器
        self.timer_of_graph_refresh = QTimer()
        self.timer_of_graph_refresh.timeout.connect(self.graph_refresh_show)
        self.timer_of_graph_refresh.start(50)  # 每50ms更新一次图表

        # 新增缓冲区处理定时器
        self.timer_of_buffer_process = QTimer()
        self.timer_of_buffer_process.timeout.connect(self.process_data_buffer)
        self.timer_of_buffer_process.start(50)  # 每50ms处理一次缓冲区

    # 初始化连接所有信号与槽函数
    def Init_all_functions(self):
        self.ui.Clear_Receive_Button.clicked.connect(self.Clear_Receive_Button_Task)
        self.ui.Save_Window_Button.clicked.connect(self.Save_Window_Button_Task)
        self.ui.Serial_Operate_Button.toggled.connect(self.Serial_Operate_Button_Task)
        self.ui.Serial_Choose.currentIndexChanged.connect(self.Serial_Choose_Task) #窗口选择类型控件连接到我们相应的槽函数
        self.ui.Baud_Rate.currentIndexChanged.connect(self.Baud_Rate_Task)
        self.ui.Data_Bit.currentIndexChanged.connect(self.Data_Bit_Task)
        self.ui.Stop_Bit.currentIndexChanged.connect(self.Stop_Bit_Task)
        self.ui.Check_Bit.currentIndexChanged.connect(self.Check_Bit_Task)

        self.ui.Hex_Show.toggled.connect(self.Hex_Show_Task)
        self.ui.DTR.toggled.connect(self.DTR_Task)
        self.ui.RTS.toggled.connect(self.RTS_Task)
        self.ui.Auto_Save.toggled.connect(self.Auto_Save_Task)
        self.ui.Time_Stamp.toggled.connect(self.Time_Stamp_Task)

        self.ui.Send_Button.clicked.connect(self.Send_Button_Task)
        self.ui.Clear_Send_Button.clicked.connect(self.Clear_Send_Button_Task)
        self.ui.R_num.setText(f'R: {len(self.receive_raw_data)}')
        self.ui.S_num.setText(f'S: {len(self.whole_send_text)}')

        self.ui.Time_Stamp_Time.textChanged.connect(self.Change_Time_Stamp_Task)
        self.ui.Send_Period_Time.textChanged.connect(self.Send_Period_Time_Change_Task)

        self.ui.Send_New_Line.toggled.connect(self.Send_New_Line_Task)
        self.ui.Time_Send.toggled.connect(self.Time_Send_Task)
        self.ui.Hex_Send.toggled.connect(self.Hex_Send_Task)

        self.ui.Clear_Graph.clicked.connect(self.Clear_Graph_Task)
        self.ui.Save_Graph.clicked.connect(self.Save_Graph_Task)
        self.ui.line_type.currentIndexChanged.connect(self.Line_Type_Change_Task)
        # self.ui.Show_Dot_Num.textChanged.connect(self.Show_Dot_Num_Task)


    # 初始化串口相关配置
    def Init_serial_settings(self):
        self.ser = Serial() #声明一个串口变量
        self.ser.close() #而后将窗口关闭，等待后续配置
        # 初始化串口列表
        self.update_serial_ports() #此函数用于初始化更新窗口列表

    # 串口选择槽函数
    def Serial_Choose_Task(self):
        self.ser.close()
        if self.ui.Serial_Operate_Button.isChecked() == True:  #True开启串口，false关闭串口
            self.ser = Serial(self.ui.Serial_Choose.currentText(), self.ui.Baud_Rate.currentText(), timeout=1) #读取选择的COM及其波特率读取下来，赋值到ser变量
            self.ser.open()

        else:
            self.ser.close()

    # 波特率选择槽函数
    def Baud_Rate_Task(self):
        self.ser.close()
        if self.ui.Serial_Operate_Button.isChecked() == True:
            self.ser = Serial(self.ui.Serial_Choose.currentText(), self.ui.Baud_Rate.currentText(), timeout=1)
            self.ser.open()

        else:
            self.ser.close()

    # 停止位选择槽函数
    def Stop_Bit_Task(self):
        stop_bits_map = {
            '1': STOPBITS_ONE,
            '1.5': STOPBITS_ONE_POINT_FIVE,
            '2': STOPBITS_TWO
        }

        stop_bits = self.ui.Stop_Bit.currentText()
        if stop_bits in stop_bits_map:
            self.ser.stopbits = stop_bits_map[stop_bits]

            if self.ser.is_open:
                try:
                    # 需要重新打开串口才能应用新的停止位设置
                    self.ser.close()
                    self.ser.open()
                except SerialException as e:
                    QMessageBox.warning(self.ui, "串口错误", f"无法重新打开串口: {str(e)}")
                    self.ui.Serial_Operate_Button.setChecked(False)

    # 数据位选择槽函数
    def Data_Bit_Task(self):
        self.ser.bytesize = int(self.ui.Data_Bit.currentText())

    # 校验位选择槽函数
    def Check_Bit_Task(self):
        if (self.ui.Check_Bit.currentText() == 'None'):
            self.ser.parity = 'N'
        elif (self.ui.Check_Bit.currentText() == 'Odd'):
            self.ser.parity = 'O'
        elif (self.ui.Check_Bit.currentText() == 'Even'):
            self.ser.parity = 'E'
        else:
            self.ser.parity = 'N'

    # 串口操作槽函数
    def Serial_Operate_Button_Task(self):
        if self.ui.Serial_Operate_Button.isChecked() == True:
            self.ser = Serial(self.ui.Serial_Choose.currentText(), self.ui.Baud_Rate.currentText(), timeout=1)
            self.ser.open()

        else:
            self.ser.close()

    # 保存窗口槽函数
    def Save_Window_Button_Task(self):
        # 检查是否有内容可保存
        if not self.receive_raw_data:
            QMessageBox.information(self.ui, "提示", "接收窗口没有数据可保存")
            return

        # 弹出文件选择对话框
        file_path, _ = QFileDialog.getSaveFileName(  #使用QFileDialog，此函数返回两个参数，第二个不需要，使用 _ 取出接收，第一个file_path为保存路径
            self.ui,
            "保存接收数据",
            f"serial_data_{datetime.now().strftime('%Y%m%d_%H%M%S')}.txt",
            "文本文件 (*.txt)"
        )

        # 如果用户选择了文件路径
        if file_path:
            try:
                # 获取接收窗口的文本内容
                content = self.ui.receive_text.toPlainText()  #当前接手界面一段话，存储到content数组里

                # 写入文件
                with open(file_path, 'w', encoding='utf-8') as f:
                    f.write(content)  #使用文件操作显示写到file_path路径txt文件里

                # 显示成功消息
                QMessageBox.information(self.ui, "成功", f"数据已成功保存到:\n{file_path}")

            except Exception as e:
                # 显示错误消息
                QMessageBox.critical(self.ui, "错误", f"保存文件时发生错误:\n{str(e)}")

    # 清除接收槽函数
    def Clear_Receive_Button_Task(self):
        self.ui.receive_text.clear()
        self.receive_raw_data.clear()
        self.ui.R_num.setText(f'R: {len(self.receive_raw_data)}')

    # 16进制显示槽函数
    def Hex_Show_Task(self):
        if self.ui.Hex_Show.isChecked() == True:
            self.is_hex_receive = True
        else:
            self.is_hex_receive = False
        self.update_receive_display()

    # DTR选择槽函数
    def DTR_Task(self):
        if self.ser.is_open:
            try:
                # 设置DTR信号状态
                self.ser.dtr = self.ui.DTR.isChecked()

            except SerialException as e:
                QMessageBox.warning(self.ui, "串口错误", f"设置DTR信号时发生错误: {str(e)}")
        else:
            self.ui.DTR.setChecked(False)

    # RTS选择槽函数
    def RTS_Task(self):
        if self.ser.is_open:
            try:
                # 设置RTS信号状态
                self.ser.rts = self.ui.RTS.isChecked()

            except SerialException as e:
                QMessageBox.warning(self.ui, "串口错误", f"设置RTS信号时发生错误: {str(e)}")
        else:
            self.ui.RTS.setChecked(False)

    # 自动选择槽函数
    def Auto_Save_Task(self):
        if self.ui.Auto_Save.isChecked():
            # 启用自动保存，第一次文件创建做准备
            if not hasattr(self, 'auto_save_file'):
                # 如果还没有设置保存文件，弹出文件选择对话框
                file_path, _ = QFileDialog.getSaveFileName(
                    self.ui,
                    "选择自动保存文件",
                    f"serial_auto_save_{datetime.now().strftime('%Y%m%d_%H%M%S')}.txt",
                    "文本文件 (*.txt)"
                )

                if file_path:
                    self.auto_save_file = file_path
                    # 立即保存当前接收到的所有数据
                    QMessageBox.warning(self.ui, "提示", "已开启自动保存！")
                    if hasattr(self, 'receive_raw_data'):
                        with open(self.auto_save_file, 'wb') as f:
                            f.write(self.receive_raw_data)
            else:
                if hasattr(self, 'receive_raw_data'):
                    with open(self.auto_save_file, 'wb') as f:
                        f.write(self.receive_raw_data)
        else:
            pass
            return

    # 自动保存定时任务
    def timer_of_auto_save_task(self):
        if hasattr(self, 'auto_save_file'): #每个时间段进行判断，自动实现保存任务
            self.Auto_Save_Task()

    # 修改时间戳周期的函数
    def Change_Time_Stamp_Task(self):
        if self.is_time_stamp:
            if self.ui.Time_Stamp_Time.displayText():
                self.time_stamp_time = int(self.ui.Time_Stamp_Time.displayText())
            else:
                self.time_stamp_time = 1000
            self.timer_of_time_stamp.stop()
            self.timer_of_time_stamp.setInterval(self.time_stamp_time)
            self.timer_of_time_stamp.start()
        else:
            pass

    # 时间戳定时任务
    def time_stamp_time_task(self):
        if self.is_time_stamp:
            if self.is_hex_receive:
                self.receive_raw_data += ('\n' + f"{datetime.now().strftime('时间戳：[%Y-%m-%d %H:%M:%S]')}" + '\n')
            else:
                self.receive_raw_data += (
                        '\n' + f"{datetime.now().strftime('时间戳：[%Y-%m-%d %H:%M:%S]')}" + '\n').encode('utf-8',
                                                                                                         errors='replace')

    # 开启时间戳槽函数
    def Time_Stamp_Task(self):
        if self.ui.Time_Stamp.isChecked() == True:
            self.is_time_stamp = True
        else:
            self.is_time_stamp = False
        self.Change_Time_Stamp_Task()

                hex_send_text += '\n' #发送新行，在最后加一个换行符

             #如果是正常文本发送

        self.ui.S_num.setText('S:' + str(len(self.whole_send_text)))    #将发送的文本都存到whole_send_text，这样保存发送数据的时候可以调用whole_send_text字符串

    def update_serial_ports(self):

        # 获取当前选择的串口（如果有COM）
        current_selection = self.ui.Serial_Choose.currentText() if self.ui.Serial_Choose.count() > 0 else ""

        # 清空当前列表

        # 添加可用串口到下拉框
        for port in ports:
            self.ui.Serial_Choose.addItem(port.device) #将每个串口更新对应到QT列表里

        # 尝试恢复之前的选择（如果仍然存在）防止出错配置
        if current_selection in self.current_ports:
            index = self.ui.Serial_Choose.findText(current_selection)
            if index >= 0:
                self.ui.Serial_Choose.setCurrentIndex(index)
                if index > 0:
                    self.Serial_Choose_Task()
        elif self.current_ports:
            # 如果没有之前的选项，但有可用串口，选择第一个
            self.ui.Serial_Choose.setCurrentIndex(0)
        # 如果没有可用串口，显示提示

    # 串口接收任务，每10ms进行一次
    def receive_serial_data_task(self):
        if self.ser and self.ser.is_open:
            try:
                self.R_cnt = self.ser.in_waiting 
                if self.R_cnt > 0:  #接收到的字节数大于零，进行一次读取
                    # 读取原始数据
                    receive_data = self.ser.read(self.R_cnt)

                    # 保存原始数据
                    self.receive_raw_data.extend(receive_data)

                    # 将数据添加到缓冲区进行处理
                    self.add_to_data_buffer(receive_data)

                    # 更新显示
                    self.update_receive_display()

                    # 更新接收计数
                    self.ui.R_num.setText(f'R: {len(self.receive_raw_data)}')

            except SerialException as e:
                QMessageBox.warning(self.ui, "串口错误", f"读取串口数据时发生错误: {str(e)}")  #如果出错，使用QMessageBox发送这个警告
                self.ser.close()
                self.ui.Serial_Operate_Button.setChecked(False)

    def add_to_data_buffer(self, data):
        try:
            # 将二进制数据转换为字符串
            text_data = data.decode('utf-8', errors='ignore')

            # 添加到缓冲区
            self.data_buffer += text_data
            self.data_points_received += 1

            # 限制缓冲区大小，防止内存溢出
            if len(self.data_buffer) > self.buffer_max_size:
            # 保留最近的数据，丢弃旧数据
                self.data_buffer = self.data_buffer[-self.buffer_max_size // 2:]

        except Exception as e:
            print(f"添加到缓冲区错误: {e}")

    def process_data_buffer(self):
        try:
            # 如果缓冲区为空，直接返回
            if not self.data_buffer:
                return

            # 查找缓冲区中所有可能的数据格式
            pattern = {
                0:r'y:(\d+)',
                1:r'x:(\d+),y:(\d+)',
                2:r'x:(\d+),y:(\d+)'
            }
            found_points = []
            if self.ui.line_type.currentIndex() == 0:
                matches = re.finditer(pattern[self.ui.line_type.currentIndex()], self.data_buffer)
                for match in matches:
                    y = float(match.group(1))
                    x = 0.0
                    found_points.append((x, y, match.start(), match.end()))
            else:
                matches = re.finditer(pattern[self.ui.line_type.currentIndex()], self.data_buffer)
                for match in matches:
                    x = float(match.group(1))
                    y = float(match.group(2))
                    found_points.append((x, y, match.start(), match.end()))

            # 如果有找到数据点，处理它们
            if found_points:
                # 按位置排序，从前往后处理
                found_points.sort(key=lambda x: x[2])

                # 取最后一个有效匹配的位置，用于更新缓冲区
                last_end_pos = 0

                for x, y, start, end in found_points:
                    # 将数据点放入队列
                    self.graph_data_queue.put((x, y))
                    self.data_points_parsed += 1
                    last_end_pos = end

                # 更新缓冲区，移除已处理的数据
                if last_end_pos > 0:
                    self.data_buffer = self.data_buffer[last_end_pos:]


        except Exception as e:
            print(f"处理缓冲区错误: {e}")

    def graph_refresh_show(self):
        try:
            # 处理队列中的所有数据点
            points_added = 0
            while not self.graph_data_queue.empty():
                try:
                    x, y = self.graph_data_queue.get_nowait()
                    global calls
                    calls += 1
                    if self.ui.line_type.currentIndex() == 0:
                        t = calls
                        y = math.sin(t * 0.2)
                    elif self.ui.line_type.currentIndex() == 1:
                        t = calls
                        x = math.sin(t*0.2)
                        y = math.cos(t*0.2)
                    elif self.ui.line_type.currentIndex() == 2:
                        t = calls
                    self.t_list.append(t)
                    self.x_list.append(x)
                    self.y_list.append(y)
                    points_added += 1
                except queue.Empty:
                    break

            # 如果添加了新的数据点，更新图表
            if points_added > 0 or len(self.x_list) > 0:
                # 限制数据点总量，防止内存溢出
                if int(self.ui.Show_Dot_Num.displayText())>1024:
                    self.ui.Show_Dot_Num.setText(1024)
                self.max_data_points = int(self.ui.Show_Dot_Num.displayText())
                if len(self.x_list) > self.max_data_points:
                    # 保留最新的数据，丢弃旧数据
                    self.x_list = self.x_list[-self.max_data_points:]
                    self.y_list = self.y_list[-self.max_data_points:]
                    self.t_list = self.t_list[-self.max_data_points:]


                line_style_map = {
                    0:'-',
                    1:'--',
                    2:'.'
                }
                line_color_map = {
                    0:'k',
                    1:'r',
                    2:'b'
                }
                # 更新图表
                self.axes.clear()
                if self.ui.line_type.currentIndex() == 0:
                    # print(self.t_list)
                    # print(self.y_list)
                    if len(self.t_list) > 1:
                        self.axes.plot(self.t_list, self.y_list,
                                       line_style_map[self.ui.line_style.currentIndex()] + line_color_map[
                                           self.ui.line_color.currentIndex()], linewidth=2)

                        self.axes.plot(self.t_list, self.y_list,
                                       line_style_map[self.ui.line_style.currentIndex()] + 'o', markersize=5)

                    self.axes.grid(True, alpha=0.3)

                    # 平移显示逻辑
                    if self.t_list and self.y_list:

                        # 如果数据点很多，我们只显示最新的window_size个x值范围

                            t_min = t_max - window_size

                            y_max = max(visible_y)
